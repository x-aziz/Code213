Exercice : Programme en C

Code C pour remplir et sommer une matrice

#include <stdio.h>

#define MAX_ROWS  3
#define MAX_COLS  3

// Fonction pour remplir la matrice
void remplirMatrice(int mat[MAX_ROWS][MAX_COLS]) {
  printf("Entrez les éléments de la matrice (%dx%d) :\n", MAX_ROWS, MAX_COLS);
  for (int i = 0; i < MAX_ROWS; i++) {
      for (int j = 0; j < MAX_COLS; j++) {
          printf("Element [%d][%d] : ", i, j);
          scanf("%d", &mat[i][j]);
      }
  }
}

// Fonction pour calculer la somme des éléments positifs
int sommePositifs(int mat[MAX_ROWS][MAX_COLS]) {
  int somme = 0;
  for (int i = 0; i < MAX_ROWS; i++) {
      for (int j = 0; j < MAX_COLS; j++) {
          if (mat[i][j] > 0) {
              somme += mat[i][j];
          }
      }
  }
  return somme;
}

int main() {
  int matrice[MAX_ROWS][MAX_COLS];
  remplirMatrice(matrice);

  // Affichage de la matrice
  printf("\nMatrice saisie :\n");
  for (int i = 0; i < MAX_ROWS; i++) {
      for (int j = 0; j < MAX_COLS; j++) {
          printf("%d\t", matrice[i][j]);
      }
      printf("\n");
  }

  // Calcul et affichage de la somme des éléments positifs
  int somme = sommePositifs(matrice);
  printf("\nLa somme des éléments positifs de la matrice est : %d\n", somme);

  return 0;
}
